    <div className="max-w-3xl mx-auto text-center">
      <h3 className="text-4xl font-semibold text-primary-600 mb-6">
        Ne restez pas dans le doute
      </h3>
      <p className="text-xl text-neutral-900 mb-12">
        Le quiz vous a donné une première indication. 
        Pour aller plus loin, échangeons de vive voix sur votre situation.
      </p>
      
      <div className="bg-white rounded-2xl shadow-xl p-8">
        <h4 className="text-2xl font-semibold text-accent-500 mb-4">
          30 minutes pour faire le point ensemble
        </h4>
        <p className="text-lg text-neutral-900 mb-8">
          Un premier échange gratuit et sans engagement pour :
        </p>
        <ul className="text-left max-w-md mx-auto space-y-3 mb-8">
          <li className="flex items-start gap-3">
            <span className="text-neutral-900">Analyser ensemble vos résultats du quiz</span>
          </li>
          <li className="flex items-start gap-3">
            <span className="text-neutral-900">Comprendre vos besoins spécifiques</span>
          </li>
          <li className="flex items-start gap-3">
            <span className="text-neutral-900">Vous présenter nos solutions d'accompagnement</span>
          </li>
          <li className="flex items-start gap-3">
            <span className="text-neutral-900">Répondre à toutes vos questions</span>
          </li>
        </ul>
        
        <button className="inline-flex px-8 py-4 rounded-full bg-accent-500 hover:bg-accent-600 text-white font-semibold shadow-lg transition text-lg">
          Réserver mon RDV découverte
        </button>
      </div>
    </div>
  </div>
</section>

<script>
{`
  // Quiz functionality
  document.addEventListener('DOMContentLoaded', function() {
    let currentQuestion = 1;
    const totalQuestions = 7;
    let score = 0;

    const questions = document.querySelectorAll('.quiz-question');
    const prevBtn = document.getElementById('prev-btn');
    const nextBtn = document.getElementById('next-btn');
    const submitBtn = document.getElementById('submit-btn');
    const restartBtn = document.getElementById('restart-quiz');
    const resultsDiv = document.getElementById('quiz-results');

    // Handle radio button changes
    document.querySelectorAll('input[type="radio"]').forEach(radio => {
      radio.addEventListener('change', function() {
        nextBtn.disabled = false;
      });
    });

    // Next button
    nextBtn.addEventListener('click', function() {
      if (currentQuestion < totalQuestions) {
        questions[currentQuestion - 1].classList.add('hidden');
        currentQuestion++;
        questions[currentQuestion - 1].classList.remove('hidden');
        
        prevBtn.classList.remove('hidden');
        nextBtn.disabled = !document.querySelector(\`input[name="q\${currentQuestion}"]:checked\`);
        
        if (currentQuestion === totalQuestions) {
          nextBtn.classList.add('hidden');
          submitBtn.classList.remove('hidden');
        }
      }
    });

    // Previous button
    prevBtn.addEventListener('click', function() {
      if (currentQuestion > 1) {
        questions[currentQuestion - 1].classList.add('hidden');
        currentQuestion--;
        questions[currentQuestion - 1].classList.remove('hidden');
        
        if (currentQuestion === 1) {
          prevBtn.classList.add('hidden');
        }
        
        nextBtn.classList.remove('hidden');
        submitBtn.classList.add('hidden');
        nextBtn.disabled = false;
      }
    });

    // Submit button
    submitBtn.addEventListener('click', function() {
      // Calculate score
      score = 0;
      for (let i = 1; i <= totalQuestions; i++) {
        const checked = document.querySelector(\`input[name="q\${i}"]:checked\`);
        if (checked) {
          score += parseInt(checked.value);
        }
      }

      // Hide quiz, show results
      document.getElementById('quiz-container').querySelector('.quiz-question[data-question="7"]').classList.add('hidden');
      submitBtn.classList.add('hidden');
      prevBtn.classList.add('hidden');
      resultsDiv.classList.remove('hidden');

      // Show appropriate result
      document.querySelectorAll('#result-high, #result-medium, #result-low').forEach(el => {
        el.classList.add('hidden');
      });

      if (score >= 15) {
        document.getElementById('result-high').classList.remove('hidden');
      } else if (score >= 8) {
        document.getElementById('result-medium').classList.remove('hidden');
      } else {
        document.getElementById('result-low').classList.remove('hidden');
      }
    });

    // Restart button
    restartBtn.addEventListener('click', function() {
      currentQuestion = 1;
      score = 0;
      
      // Reset all radios
      document.querySelectorAll('input[type="radio"]').forEach(radio => {
        radio.checked = false;
      });
      
      // Hide all questions
      questions.forEach(q => q.classList.add('hidden'));
      
      // Show first question
      questions[0].classList.remove('hidden');
      
      // Reset buttons
      prevBtn.classList.add('hidden');
      nextBtn.classList.remove('hidden');
      nextBtn.disabled = true;
      submitBtn.classList.add('hidden');
      resultsDiv.classList.add('hidden');
    });
  });
`}
</script>
